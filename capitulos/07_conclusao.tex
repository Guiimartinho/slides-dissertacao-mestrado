% --- Slide 36: Limitações e Trabalhos Futuros ---
\begin{frame}{Limitações e Trabalhos Futuros}
  \vfill
  \begin{columns}[t]
    \begin{column}{0.46\textwidth}
      {\normalsize\textcolor{skipcolor}{\textbf{Limitações}}}
      \vspace{3mm}
      \begin{itemize}\setlength{\itemsep}{3mm}
        \item \textbf{Propagação de erros}\\
              {\scriptsize PGIE limita SGIEs: cobertura\\efetiva cai para 0,59--0,76}
        \item \textbf{Objetos pequenos}\\
              {\scriptsize Ferramentas manuais com\\AP < 0,60 (Cutter, Trowel)}
        \item \textbf{Validação controlada}\\
              {\scriptsize Benchmark, sem canteiro real\\com poeira, chuva, iluminação}
        \item \textbf{Dataset sem dados locais}\\
              {\scriptsize Canteiros brasileiros não\\representados no treinamento}
        \item \textbf{Desbalanceamento}\\
              {\scriptsize NO-Safety Vest: AP=0,236\\com apenas 890 instâncias}
      \end{itemize}
    \end{column}
    \begin{column}{0.50\textwidth}
      {\normalsize\textcolor{headcolor}{\textbf{Trabalhos Futuros}}}
      \vspace{3mm}
      \begin{itemize}\setlength{\itemsep}{3mm}
        \item \textcolor{headcolor}{$\boldsymbol{\checkmark}$} \textbf{SAHI} para objetos pequenos\\
              {\scriptsize Slicing Aided Hyper Inference:\\melhora detecção a distância}
        \item \textcolor{headcolor}{$\boldsymbol{\checkmark}$} \textbf{Dataset brasileiro}\\
              {\scriptsize Coleta em canteiros reais com\\foco em classes sub-representadas}
        \item \textcolor{headcolor}{$\boldsymbol{\checkmark}$} \textbf{YOLOv9 / YOLOv10 / RT-DETR}\\
              {\scriptsize Arquiteturas mais recentes com\\melhor trade-off precisão/velocidade}
        \item \textcolor{headcolor}{$\boldsymbol{\checkmark}$} \textbf{Piloto em canteiro real}\\
              {\scriptsize Validação com profissionais de\\segurança do trabalho (NR-18)}
        \item \textcolor{headcolor}{$\boldsymbol{\checkmark}$} \textbf{Quantização INT8 + DLA}\\
              {\scriptsize Potencial para $>$30 FPS com\\acelerador dedicado do Jetson}
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
\end{frame}
