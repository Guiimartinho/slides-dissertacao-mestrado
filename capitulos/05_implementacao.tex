    % Routing conditions
    \node[routebox, right=1.5cm of pgie, yshift=1.4cm] (r0) {class\_id == 0\\(Person)};
    \node[routebox, right=1.5cm of pgie] (r1) {class\_id == 1\\(Vehicle)};
    \node[routebox, right=1.5cm of pgie, yshift=-1.4cm] (r4) {class\_id == 4\\(Tool)};

    \node[sgiebox, right=1.2cm of r0] (sgie1) {\textbf{SGIE-EPI}\\{\footnotesize 14 classes}};

    % Arrows PGIE -> routing
    \draw[myarrow] (pgie.east) -- ++(0.3,0) |- (r0.west);
    \draw[myarrow] (pgie.east) -- (r1.west);
    \draw[myarrow] (pgie.east) -- ++(0.3,0) |- (r4.west);

    % Arrows routing -> SGIEs
    \draw[myarrow, draw=detectioncolor] (r0.east) -- node[above, font=\tiny, text=detectioncolor] {crop ROI} (sgie1.west);
    \draw[myarrow, draw=detectioncolor] (r1.east) -- node[above, font=\tiny, text=detectioncolor] {crop ROI} (sgie2.west);
    \draw[myarrow, draw=detectioncolor] (r4.east) -- node[above, font=\tiny, text=detectioncolor] {crop ROI} (sgie3.west);

  \end{tikzpicture}

  \vspace{2mm}

  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{\footnotesize Função do Parser}
        \begin{itemize}\scriptsize
          \item Decodifica saída \textit{anchor-free} YOLOv8
          \item Filtragem por confiança + NMS
          \item Compilado como \texttt{.so} (C++)
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{\footnotesize Roteamento}
        \begin{itemize}\scriptsize
          \item Cada SGIE opera via \texttt{operate-on-class-ids}
          \item ROI recortada da bbox do PGIE
          \item SGIEs processam apenas regiões de interesse
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

% --- Slide 24: Deployment (NOVO) ---
\begin{frame}{Deployment na Borda}
  \smallskip
  \centering
  {\small Aspectos práticos de implantação do sistema no Jetson AGX Xavier.}

  \vspace{3mm}

  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Configuração do Dispositivo}
        \begin{itemize}\small
          \item Modo de potência: \textbf{MAXN (30\,W)}
          \item 512 CUDA cores ativos
          \item 8 cores ARM v8.2 a 2,26 GHz
          \item 32 GB LPDDR4x compartilhada
          \item Refrigeração passiva + fan
        \end{itemize}
      \end{block}

      \vspace{2mm}

      \begin{block}{\small Entrada de Vídeo}
        \begin{itemize}\small
          \item Stream \textbf{RTSP} via rede local
          \item Suporte a arquivo local (MP4/H.264)
          \item Decodificação por \textbf{NVDEC} (hardware)
          \item Resolução: 1920$\times$1080 $\to$ 640$\times$640
        \end{itemize}
      \end{block}
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{block}{\small Saída e Visualização}
        \begin{itemize}\small
          \item Display local via \texttt{nveglglessink}
          \item Gravação em arquivo via \texttt{filesink}
          \item Anotações: bounding boxes + labels
          \item Alertas de violação de EPI
        \end{itemize}
      \end{block}

      \vspace{2mm}

      \begin{block}{\small Otimizações Aplicadas}
        \begin{itemize}\small
          \item Memória unificada CPU-GPU
          \item Batching de crops para SGIEs
          \item Pipeline assíncrono (GStreamer)
          \item Quantização FP16 em todos os modelos
          \item Votação temporal para reduzir FP
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vfill

  \begin{center}
    \footnotesize
    \textcolor{labelcolor}{Pipeline completo: \textbf{20 FPS} / \textbf{58 ms} de latência end-to-end em modo MAXN.}
  \end{center}

  \vfill
\end{frame}
