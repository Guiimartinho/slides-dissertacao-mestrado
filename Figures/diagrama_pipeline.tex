% ============================================================================
% Diagrama do Pipeline GStreamer/DeepStream
% Layout compacto: Row 1 horizontal, SGIEs empilhados verticalmente
% ============================================================================

\begin{tikzpicture}[
    node distance=5mm,
    block/.style={
        rectangle, rounded corners=3pt,
        minimum width=20mm, minimum height=11mm,
        align=center, font=\small\sffamily\bfseries,
        text=white, draw=none
    },
    sblock/.style={
        rectangle, rounded corners=2pt,
        minimum width=22mm, minimum height=8mm,
        align=center, font=\footnotesize\sffamily\bfseries,
        text=white, draw=none
    },
    arr/.style={->, >=Stealth, line width=0.8pt, draw=black!60},
    plbl/.style={font=\tiny\ttfamily, text=black!50},
]

% ============================================================================
% Row 1: Source -> Decoder -> Streammux -> PGIE -> Tracker
% ============================================================================

\node[block, fill=teal!70!black] (src) {Source\\[-1pt]{\tiny Camera/File}};
\node[block, fill=cyan!65!black, right=of src] (dec) {Decoder\\[-1pt]{\tiny NVDEC}};
\node[block, fill=blue!65!black, right=of dec] (mux) {Streammux\\[-1pt]{\tiny Batching}};
\node[block, fill=purple!65!black, right=of mux] (pgie) {PGIE\\[-1pt]{\tiny nvinfer}};
\node[block, fill=violet!65!black, right=of pgie] (trk) {Tracker\\[-1pt]{\tiny NvDCF}};

% Row 1 arrows
\draw[arr] (src) -- (dec);
\draw[arr] (dec) -- (mux);
\draw[arr] (mux) -- (pgie);
\draw[arr] (pgie) -- (trk);

% ============================================================================
% Row 2: SGIEs (stacked vertically) + OSD + Sink
% ============================================================================

% SGIEs stacked vertically below PGIE/Tracker area
\node[sblock, fill=orange!70!black, below=12mm of pgie] (s1) {SGIE-1 (EPI)};
\node[sblock, fill=orange!70!black, below=1.5mm of s1] (s2) {SGIE-2 (Maq)};
\node[sblock, fill=orange!70!black, below=1.5mm of s2] (s3) {SGIE-3 (Fer)};

% Container around SGIEs
\node[draw=orange!50, dashed, rounded corners=3pt, inner sep=5pt,
      fit=(s1)(s2)(s3),
      label={[font=\small\sffamily\bfseries, orange!70!black]above:SGIEs}] (sg) {};

% OSD and Sink
\node[block, fill=green!55!black, right=10mm of sg.east, anchor=west] (osd) {OSD\\[-1pt]{\tiny nvdsosd}};
\node[block, fill=red!55!black, right=of osd] (sink) {Sink\\[-1pt]{\tiny Output}};

% ============================================================================
% Connecting arrows between rows
% ============================================================================

% Tracker down to SGIE group
\draw[arr] (trk.south) -- ++(0,-4mm) -| (sg.north);

% SGIEs to OSD
\draw[arr] (sg.east) -- (osd.west);

% OSD to Sink
\draw[arr] (osd) -- (sink);

\end{tikzpicture}
